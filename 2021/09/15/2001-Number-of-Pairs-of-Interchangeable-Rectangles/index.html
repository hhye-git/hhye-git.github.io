2001. Number of Pairs of Interchangeable Rectangles

發表於 2021-09-15

解1.
import java.math.BigInteger;

class Solution {

    public long interchangeableRectangles(int[][] rectangles) {
        long count = 0;
        HashMap<String, Integer> map = new HashMap<>();
        for (int[] rectangle : rectangles) {
            int gcd = BigInteger.valueOf(rectangle[0]).gcd(BigInteger.valueOf(rectangle[1])).intValue();
            String key = rectangle[0] / gcd + " " + rectangle[1] / gcd;
            count += map.getOrDefault(key, 0);
            map.put(key, 1 + map.getOrDefault(key, 0));
        }
        return count;
    }
}

class Solution {
public long interchangeableRectangles(int[][] rectangles) {
Comparator<int[]> c = new Comparator<int[]>(){
public int compare(int[] a, int[] b){
return Long.compare((long)a[0] * b[1], (long)b[0] * a[1]);
}
};
Arrays.sort(rectangles, c);
int times = 0;
long ans = 0;
for(int i = 0; i < rectangles.length; i++){
if(i != rectangles.length-1 && c.compare(rectangles[i], rectangles[i+1]) == 0){
times++;
ans += times;
}
else{
times = 0;
}
}
return ans;
}
}
